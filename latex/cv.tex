%%%%%%%%%%%%%%%%%%%%%%%% Comandos Facilitadores %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%% Cabecalho, com uma linha horizontal logo abaixo
\newcommand{\makeheading}[1]%
        {\hspace*{-\marginparsep minus \marginparwidth}%
         \begin{minipage}[t]{\textwidth+\marginparwidth+\marginparsep}%
                {\large \bfseries #1}\\[-0.15\baselineskip]%
                 \rule{\columnwidth}{1pt}%
         \end{minipage}}

% Cabecalhos de Secao
%
% Uso: \section{nome da secao}
%
% Importante, deixe o texto logo abaixo da linha que indica a secao, para fins
% de alinhamento. Ex:
%
%       \section{Secao}
%       Texto do inicio da secao.
%
%% OU
%       \section{Secao}
%       %
%       Texto do inicio da secao.
%
%% Mas nao:
%
%       \section{Secao}
%
%       Texto do inicio da secao.
%
\renewcommand{\section}[2]%
        {\pagebreak[2]\vspace{1.3\baselineskip}%
         \phantomsection\addcontentsline{toc}{section}{#1}%
         \hspace{0in}%
         \marginpar{
         \raggedright \scshape #1}#2}

%% Uma lista do tipo itemize com bastante espaco entre os itens
\newenvironment{outerlist}[1][\enskip\textbullet]%
        {\begin{itemize}[#1]}{\end{itemize}%
         \vspace{-1em}}

\newenvironment{compactouterlist}[1][\enskip\textbullet]%
        {\begin{itemize} \setlength{\itemsep}{0.1em}}{\end{itemize}%
         \vspace{-1em}}

%% Criando um ambiente identico aa lista externa que possui melhor espacamento
% quando usado no inicio de uma \section
\newenvironment{lonelist}[1][\enskip\textbullet]%
        {\vspace{-\baselineskip}\begin{list}{#1}{%
        \setlength{\partopsep}{0pt}%
        \setlength{\topsep}{0pt}}}
        {\end{list}\vspace{-.6\baselineskip}}

%% Uma lista estilo itemize com pouco espaco entre os itens
\newenvironment{innerlist}[1][\enskip\textbullet]%
        {\begin{compactitem}[#1]}{\end{compactitem}}

%% Criando um ambiente identico aa lista interna que possui melhor espacamento
% quando usado no inicio de uma \section
\newenvironment{loneinnerlist}[1][\enskip\textbullet]%
        {\vspace{-\baselineskip}\begin{compactitem}[#1]}
        {\end{compactitem}\vspace{-.6\baselineskip}}

% Para adicionar algum espaco entre paragrafos
% Isso tb informa ao LaTeX para pular para a proxima pagina neste ponto.
\newcommand{\blankline}{\quad\pagebreak[2]}

% Hiperlinks para documentos DOI
\newcommand\doilink[1]{\href{http://dx.doi.org/#1}{#1}}
\newcommand\doi[1]{doi:\doilink{#1}}
